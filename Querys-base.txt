-- ── Paso 2: Tablas principales ───────────────────────────────────────────────
-- RF_11: Creación de tablas

-- Usuarios (RF_01_seg, RF_02_seg, RF_04_seg, RF_05_seg, RF_07_seg–RF_13_seg)
CREATE TABLE users (
    id           SERIAL       PRIMARY KEY,
    username     VARCHAR(50)  UNIQUE NOT NULL,
    email        VARCHAR(100) UNIQUE NOT NULL,
    password     VARCHAR(255) NOT NULL,
    gender       VARCHAR(20),
    role         VARCHAR(20)  DEFAULT 'user',
    verified     BOOLEAN      DEFAULT FALSE,
    blocked      BOOLEAN      NOT NULL DEFAULT FALSE,
    pending_email VARCHAR(255),
    created_at   TIMESTAMP    DEFAULT NOW(),

    CONSTRAINT users_gender_check
        CHECK (gender IN ('masculino', 'femenino', 'prefiero_no_decir') OR gender IS NULL)
);

-- Escuelas (RF_03: filtrar por escuela)
CREATE TABLE schools (
    id   SERIAL       PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    city VARCHAR(100)
);

-- Stickers georeferenciados (RF_01–RF_09, RF_12)
CREATE TABLE stickers (
    id         SERIAL      PRIMARY KEY,
    user_id    INTEGER     REFERENCES users(id)   ON DELETE SET NULL,
    school_id  INTEGER     REFERENCES schools(id) ON DELETE SET NULL,
    category   VARCHAR(50) NOT NULL,
    geojson    JSONB       NOT NULL,
    location   GEOMETRY(Point, 4326),
    created_at TIMESTAMP   DEFAULT NOW(),

    CONSTRAINT stickers_category_check
        CHECK (category IN ('transito', 'recreacion', 'riesgo', 'afecto'))
);

-- OTP de dos factores (RF_02_seg, RF_12_seg: verificación de correo)
CREATE TABLE auth_otp_challenges (
    id           SERIAL      PRIMARY KEY,
    challenge_id UUID        UNIQUE NOT NULL,
    user_id      INTEGER     NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    code_hash    VARCHAR(255) NOT NULL,
    expires_at   TIMESTAMP   NOT NULL,
    attempts     INTEGER     DEFAULT 0,
    consumed     BOOLEAN     DEFAULT FALSE,
    created_at   TIMESTAMP   DEFAULT NOW()
);

CREATE INDEX idx_auth_otp_user ON auth_otp_challenges(user_id);
CREATE INDEX idx_auth_otp_exp  ON auth_otp_challenges(expires_at);

-- Tokens de recuperación de contraseña (RF_09_seg: olvidé mi contraseña)
CREATE TABLE password_reset_tokens (
    id         SERIAL      PRIMARY KEY,
    token      VARCHAR(36) UNIQUE NOT NULL,
    user_id    INTEGER     NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    expires_at TIMESTAMP   NOT NULL,
    consumed   BOOLEAN     DEFAULT FALSE,
    created_at TIMESTAMP   DEFAULT NOW()
);

CREATE INDEX idx_prt_token   ON password_reset_tokens(token);
CREATE INDEX idx_prt_user_id ON password_reset_tokens(user_id);

-- ── Paso 4: Triggers de auditoría ────────────────────────────────────────────
-- RF_15: Trigger cuando alguien EDITA un dato
-- RF_16: Trigger cuando alguien AÑADE un dato
-- RF_17: Trigger cuando alguien ELIMINA un dato
-- RF_50: Al eliminar un usuario desde su perfil, el trigger salta

-- Tabla de auditoría central
CREATE TABLE audit_log (
    id         SERIAL       PRIMARY KEY,
    table_name VARCHAR(100) NOT NULL,
    operation  VARCHAR(10)  NOT NULL,   -- 'INSERT' | 'UPDATE' | 'DELETE'
    record_id  INTEGER,                  -- id del registro afectado
    old_data   JSONB,                    -- valores antes del cambio (UPDATE/DELETE)
    new_data   JSONB,                    -- valores después del cambio (INSERT/UPDATE)
    changed_at TIMESTAMP    NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_audit_table     ON audit_log(table_name);
CREATE INDEX idx_audit_operation ON audit_log(operation);
CREATE INDEX idx_audit_changed   ON audit_log(changed_at);


-- Función genérica reutilizada por todos los triggers de auditoría
CREATE OR REPLACE FUNCTION fn_audit_generic()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        -- RF_16: registrar inserción
        INSERT INTO audit_log (table_name, operation, record_id, new_data)
        VALUES (TG_TABLE_NAME, 'INSERT', NEW.id, row_to_json(NEW)::JSONB);
        RETURN NEW;

    ELSIF TG_OP = 'UPDATE' THEN
        -- RF_15: registrar actualización (solo si hubo cambio real)
        IF row(NEW.*) IS DISTINCT FROM row(OLD.*) THEN
            INSERT INTO audit_log (table_name, operation, record_id, old_data, new_data)
            VALUES (TG_TABLE_NAME, 'UPDATE', OLD.id,
                    row_to_json(OLD)::JSONB,
                    row_to_json(NEW)::JSONB);
        END IF;
        RETURN NEW;

    ELSIF TG_OP = 'DELETE' THEN
        -- RF_17 / RF_50: registrar eliminación
        INSERT INTO audit_log (table_name, operation, record_id, old_data)
        VALUES (TG_TABLE_NAME, 'DELETE', OLD.id, row_to_json(OLD)::JSONB);
        RETURN OLD;
    END IF;
END;
$$ LANGUAGE plpgsql;


-- Trigger sobre users (cubre RF_15, RF_16, RF_17 y RF_50)
CREATE TRIGGER trg_users_audit
AFTER INSERT OR UPDATE OR DELETE ON users
FOR EACH ROW EXECUTE FUNCTION fn_audit_generic();

-- Trigger sobre stickers (cubre RF_15, RF_16, RF_17)
CREATE TRIGGER trg_stickers_audit
AFTER INSERT OR UPDATE OR DELETE ON stickers
FOR EACH ROW EXECUTE FUNCTION fn_audit_generic();

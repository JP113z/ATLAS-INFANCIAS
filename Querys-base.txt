// comandos iniciales

CREATE DATABASE atlas_infancias;
CREATE EXTENSION IF NOT EXISTS postgis;

CREATE TABLE users (
    id         SERIAL PRIMARY KEY,
    username   VARCHAR(50)  UNIQUE NOT NULL,
    email      VARCHAR(100) UNIQUE NOT NULL,
    password   VARCHAR(255) NOT NULL,
    gender     VARCHAR(20),
    role       VARCHAR(20)  DEFAULT 'user',
    verified   BOOLEAN      DEFAULT FALSE,
    created_at TIMESTAMP    DEFAULT NOW()
);

CREATE TABLE schools (
    id   SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    city VARCHAR(100)
);

CREATE TABLE stickers (
    id         SERIAL PRIMARY KEY,
    user_id    INTEGER REFERENCES users(id)   ON DELETE SET NULL,
    school_id  INTEGER REFERENCES schools(id) ON DELETE SET NULL,
    category   VARCHAR(50) NOT NULL,
    geojson    JSONB       NOT NULL,
    location   GEOMETRY(Point, 4326),
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE votes (
    id         SERIAL PRIMARY KEY,
    sticker_id INTEGER REFERENCES stickers(id) ON DELETE CASCADE,
    user_id    INTEGER REFERENCES users(id)    ON DELETE SET NULL,
    question   TEXT    NOT NULL,
    answer     BOOLEAN NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE vote_sessions (
    id         SERIAL PRIMARY KEY,
    code       VARCHAR(10) UNIQUE NOT NULL,
    sticker_id INTEGER REFERENCES stickers(id) ON DELETE CASCADE,
    question   TEXT NOT NULL,
    active     BOOLEAN DEFAULT TRUE NOT NULL,
    created_by INTEGER REFERENCES users(id) ON DELETE SET NULL,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE vote_answers (
    id         SERIAL PRIMARY KEY,
    session_id INTEGER NOT NULL REFERENCES vote_sessions(id) ON DELETE CASCADE,
    user_id    INTEGER REFERENCES users(id) ON DELETE SET NULL,
    answer     BOOLEAN NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE (session_id, user_id)
);


-- ============================================================
-- TRIGGERS DE AUDITORÍA
-- RF_15: Trigger cuando alguien EDITA un dato
-- RF_16: Trigger cuando alguien AÑADE un dato
-- RF_17: Trigger cuando alguien ELIMINA un dato
-- RF_50: Trigger específico al eliminar un usuario desde su perfil
-- ============================================================

-- Tabla de auditoría central
CREATE TABLE audit_log (
    id          SERIAL PRIMARY KEY,
    table_name  VARCHAR(100)  NOT NULL,
    operation   VARCHAR(10)   NOT NULL,  -- 'INSERT', 'UPDATE', 'DELETE'
    record_id   INTEGER,                  -- id del registro afectado
    old_data    JSONB,                    -- valores antes del cambio (UPDATE/DELETE)
    new_data    JSONB,                    -- valores después del cambio (INSERT/UPDATE)
    changed_at  TIMESTAMP     NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_audit_table     ON audit_log(table_name);
CREATE INDEX idx_audit_operation ON audit_log(operation);
CREATE INDEX idx_audit_changed   ON audit_log(changed_at);


-- ── Función genérica de auditoría ────────────────────────────────────────────
-- Usada por los triggers de INSERT, UPDATE y DELETE en todas las tablas clave.
CREATE OR REPLACE FUNCTION fn_audit_generic()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        -- RF_16: registro de inserción
        INSERT INTO audit_log (table_name, operation, record_id, new_data)
        VALUES (TG_TABLE_NAME, 'INSERT', NEW.id, row_to_json(NEW)::JSONB);
        RETURN NEW;

    ELSIF TG_OP = 'UPDATE' THEN
        -- RF_15: registro de actualización (solo si hubo cambios reales)
        IF row(NEW.*) IS DISTINCT FROM row(OLD.*) THEN
            INSERT INTO audit_log (table_name, operation, record_id, old_data, new_data)
            VALUES (TG_TABLE_NAME, 'UPDATE', OLD.id,
                    row_to_json(OLD)::JSONB,
                    row_to_json(NEW)::JSONB);
        END IF;
        RETURN NEW;

    ELSIF TG_OP = 'DELETE' THEN
        -- RF_17: registro de eliminación
        INSERT INTO audit_log (table_name, operation, record_id, old_data)
        VALUES (TG_TABLE_NAME, 'DELETE', OLD.id, row_to_json(OLD)::JSONB);
        RETURN OLD;
    END IF;
END;
$$ LANGUAGE plpgsql;


-- ── Triggers sobre la tabla users ────────────────────────────────────────────
-- Cubre RF_15 (editar usuario), RF_16 (crear usuario),
--       RF_17 (eliminar usuario), RF_50 (eliminación desde perfil del mismo usuario)

CREATE TRIGGER trg_users_audit
AFTER INSERT OR UPDATE OR DELETE ON users
FOR EACH ROW EXECUTE FUNCTION fn_audit_generic();


-- ── Triggers sobre la tabla stickers ─────────────────────────────────────────
-- Cubre RF_15, RF_16 y RF_17 para stickers

CREATE TRIGGER trg_stickers_audit
AFTER INSERT OR UPDATE OR DELETE ON stickers
FOR EACH ROW EXECUTE FUNCTION fn_audit_generic();


-- ── Triggers sobre sticker_comments ──────────────────────────────────────────
-- Cubre RF_15, RF_16 y RF_17 para comentarios

CREATE TRIGGER trg_comments_audit
AFTER INSERT OR UPDATE OR DELETE ON sticker_comments
FOR EACH ROW EXECUTE FUNCTION fn_audit_generic();


-- ── Triggers sobre vote_sessions y vote_answers ───────────────────────────────
-- Cubre RF_15, RF_16 y RF_17 para el módulo de votaciones

CREATE TRIGGER trg_vote_sessions_audit
AFTER INSERT OR UPDATE OR DELETE ON vote_sessions
FOR EACH ROW EXECUTE FUNCTION fn_audit_generic();

CREATE TRIGGER trg_vote_answers_audit
